34  /**
35   * Our own custom resolver which knows to go looking for include files (*.inc) as resources, if
36   * a flag is set.
37   * <p>
38   * Note that this overrides {@link XMLCatalogResolver} and that we still (for now) rely on the
39   * older catalog functionality to work with the underlying schemas. We should be able to extend
40   * this resolver to locate those schemas (as resources), and move away from the write-to-disk-
41   * and-catalog functionality completely. TODO
42   */
43  public class InterceptingResolver extends XMLCatalogResolver {
44  
45  	private final Log log;
46  
47  	/**
48  	 * The resource loader can find resources in places a plain ClassLoader couldn't
49  	 */
50  	private final ResourceLoader resourceLoader;
51  
52  	public InterceptingResolver() {
53  		this(new SystemStreamLog(), null, new String[0]);
54  	}
55  
56  	/**
57  	 * @param log mojo logger
58  	 * @param resourceLoader finds resources. If null, means should be used (ie. don't do
59  	 * 		resource-based loading)
60  	 * @param catalogs catalogs to be used by underlying resolver
61  	 */
62  	public InterceptingResolver(Log log, ResourceLoader resourceLoader, String[] catalogs) {
63  		super(catalogs);
64  		for (String f : catalogs) {
65  			if (!new File(f).exists()) {
66  				throw new PluginException("Given nonexistent catalog: " + f);
67  			}
68  		}
69  		this.log = log;
70  		this.resourceLoader = resourceLoader;
71  	}
72  
73  	@Override
74      public InputSource resolveEntity (
75      			    String name,
76      			    String publicId,
77      			    String baseURI,
78      			    String systemId)
79  	throws SAXException, IOException {
80  
81  		try {
82  			if (shouldLoadAsResource(systemId)) {
83  				log.debug("Loading entity '" + systemId + "' as resource");
84  				return resourceToInputSource(publicId, systemId);
85  			}
86  			else {
87  				return super.resolveEntity(publicId, systemId);
88  			}
89  		}
90  		catch (Exception e) {
91  			// not in spec but too bad
92  			throw new PluginException("Error resolving entity: " + systemId + ": " + e, e);
93  		}
94  	}
95  
96  	@Override
97  	public LSInput resolveResource(
98  					String type,
99  					String namespaceURI,
100 					String publicId,
101 					String systemId,
102 					String baseURI) {
103 
104 		try {
105 			if (shouldLoadAsResource(systemId)) {
106 				log.debug("Loading resource '" + systemId + "' as resource");
107 				return resourceToLSInput(publicId, systemId);
108 			}
109 			else {
110                 return super.resolveResource(type, namespaceURI, publicId, systemId, baseURI);
111 			}
112 		}
113 		catch (Exception e) {
114 			// this is cheating, the spec says allow for an exception, but we don't care
115 			throw new PluginException("Error resolving resource: " + systemId + ": " + e, e);
116 		}
117 	}
118 
119 	private InputStream resourceAsStream(String systemId) {
120 
121 		String resource = resourceFromId(systemId);
122 		InputStream is = resourceLoader.getResourceAsStream(resource);
123 		if (is == null) {
124 			// worth logging since we might otherwise miss this
125 			log.warn("Resource '" + resource + "' not found.");
126 		}
127 		return is;
128 	}
129 
130 	private String resourceFromId(String systemId) {
131 		int endPos = systemId.lastIndexOf('/');
132 		return endPos > 0 ? systemId.substring(endPos) : systemId;
133 	}
134 
135 	private InputSource resourceToInputSource(String publicId, String systemId) {
136 
137 		InputStream is = resourceAsStream(systemId);
138 
139 		if (is != null) {
140 			InputSource s = new InputSource(is);
141 			s.setPublicId(publicId);
142 			s.setSystemId(systemId);
143 
144 			return s;
145 		}
146 		else {
147 			return null;
148 		}
149 	}
150 
151 	private LSInput resourceToLSInput(String publicId, String systemId) {
152 
153 		InputStream is = resourceAsStream(systemId);
154 
155 		if (is != null) {
156 			DOMInputImpl result = new DOMInputImpl(); // any old impl would do
157 			result.setByteStream(is);
158 			result.setCharacterStream(null);
159 			result.setPublicId(publicId);
160 			result.setSystemId(systemId);
161 
162 			return result;
163 		}
164 		else {
165 			return null;
166 		}
167 	}
168 
169 	private boolean shouldLoadAsResource(String systemId) {
170 		return (resourceLoader != null) && systemId != null && systemId.endsWith(".inc");
171 	}
